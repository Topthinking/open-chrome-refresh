### install
```shell
$ npm install open-chrome-refresh
```

### Usage
```js
const open = require('open-chrome-refresh');

open('https://github.com/Topthinking/open-chrome-refresh') // open
open('https://github.com/Topthinking/open-chrome-refresh') // refresh!
```
